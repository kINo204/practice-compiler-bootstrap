<program>
        : <func_def>*
        ;

<func_def>
        : <func_type> ident open_par close_par open_brac <statement> close_brac
        ;

<func_type>
        : keyword_int
        ;

<statement>
        : keyword_return <expr> semicolon
        ;

<expr>
        : <term> { ("+" | "-") <term> }
        ;

<term>
        : <factor> { ("*" | "/") <factor> }
        ;

<factor>
        : open_par <expr> close_par
        | <unary_op> <factor>
        | <literal> 
        ;

<unary_op>
        : bit_compl
        | minus
        | negation
        ;


<literal>
        : literal_int
        | literal_hex
        ;
