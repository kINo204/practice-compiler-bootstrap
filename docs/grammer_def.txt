<program>
        : <func_def>*
        ;

<func_def>
        : <func_type> ident open_par close_par open_brac <statement> close_brac
        ;

<func_type>
        : keyword_int
        ;

<statement>
        : keyword_return <exp> semicolon
        ;

<exp>
        : <logical-and-exp> { "||" <logical-and-exp> }
        ;

<logical-and-exp>
        : <equality-exp> { "&&" <equality-exp> }
        ;

<equality-exp>
        : <relational-exp> [ ("!=" | "==") <relational-exp> ]
        ;

<relational-exp>
        : <additive-exp> [ ("<" | ">" | "<=" | ">=") <additive-exp> ]
        ;

<additive_exp>
        : <term> { ("+" | "-") <term> }
        ;

<term>
        : <factor> { ("*" | "/") <factor> }
        ;

<factor>
        : open_par <exp> close_par
        | <unary_op> <factor>
        | <literal> 
        ;

<unary_op>
        : bit_compl
        | minus
        | negation
        ;


<literal>
        : literal_int
        | literal_hex
        ;
